2.2 Matrix Chart

#Carga de todas las librerías necesarias para la PEC
library(cluster)
library(tidyverse)
library(corrplot)
library(gridExtra)
library(GGally)
library(knitr)

wine_data<-read.csv("http://archive.ics.uci.edu/ml/machine-learning-databases/wine/wine.data", stringsAsFactors = FALSE, header = FALSE, strip.white = TRUE)

attach(wine_data)

# nombres de columna
colnames(wine_data) <- c("class", "alcohol", "malicAcid", "ash", "alcalinityOfAsh", "magnesium", "totalPhenols", "flavanoids", "nonflavanoidPhenols", "proanthocyanins", "colorIntensity", "hue", "dilutedWines", "proline")

colSums(is.na(wine_data)) #Comprobacion de si hay valores vacios o nulos
##               class             alcohol           malicAcid                 ash 
##                   0                   0                   0                   0 
##     alcalinityOfAsh           magnesium        totalPhenols          flavanoids 
##                   0                   0                   0                   0 
## nonflavanoidPhenols     proanthocyanins      colorIntensity                 hue 
##                   0                   0                   0                   0 
##        dilutedWines             proline 
##                   0                   0
x <- wine_data[-1]

corrplot(cor(x), type="upper", method="ellipse", tl.cex=0.9)

2.3 Área Chart

#El otro plot se usa para visualizar diagramas de densidad

x %>%
  gather(Attributes, value, 1:13) %>%
  ggplot(aes(x=value, fill=Attributes)) +
  geom_density(colour="black", alpha=0.5, show.legend=FALSE) +
  facet_wrap(~Attributes, scales="free_x") +
  labs(x="Valores", y="Densidad",
       title="Gráfica de densidad") +
  theme_bw()
